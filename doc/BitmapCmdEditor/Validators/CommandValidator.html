def validate(input)
	begin
		args=input.split(' ')
		raise NotImplementedCommand.new(ErrorMessage.new(:command_not_exist).show_content) unless BitmapCmdEditor::COMMANDS.include?(args[0])
		:valid
	rescue NotImplementedCommand => err
		err.message
	end
end
