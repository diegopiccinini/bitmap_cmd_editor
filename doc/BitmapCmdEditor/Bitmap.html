def process_command(input)

	validator=Validators::CommandValidator.validate(input)
	if validator == :valid
		args = input.split(' ')
		case args[0]
		when 'I'
			create_image(args)
		when 'C'
			clear_image(args)
		when 'L'
			colours_pixel(args)
		when 'V'
			draw_vertical_line(args)
		when 'H'
			draw_horizontal_line(args)
		when 'F'
			fill_region(args)
		when 'S'
			print_table(args)
		end
	else
		validator
	end
end
